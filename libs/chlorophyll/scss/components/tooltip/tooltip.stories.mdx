import { Meta, Canvas, Story } from '@storybook/addon-docs';

export const Template = (args) => {
  return `
        <div style="display:grid; place-items: center; height: 30rem;">
          <div>
            <button class="ghost">
              <div inert="true" role="tooltip" id="example-tool-tip" tip-position="${args.tipPosition}">${args.tipContent}</div>
              Hover me
            </button>
          </div>
        </div>
  `
}

<Meta
  title="Components/Tooltip"
  argTypes={{

/>
